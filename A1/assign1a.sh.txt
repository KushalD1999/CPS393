!/bin/bash

echo "-------------------------------------------------------------------------------"

echo $(last|head -n -2| cut -c1-9 |sort -u| uniq -u |grep -v '^reboot'| wc -l) "users have logged into" $HOSTNAME "since" $(last| tac| head -1 | cut -c13-23) "they are;"
